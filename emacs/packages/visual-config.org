(use-package perspective
  :ensure t
  :hook (after-init . persp-mode)
  :bind (("M-N" . persp-next)
         ("M-P" . persp-prev)
         ("M-J" . persp-switch))
  :init

               (dolist (k '("M-N" "M-P" "M-J"))
                 (local-set-key (kbd k) 'ignore))))
  (advice-add 'persp-switch
              :after

                  (message (persp-name (persp-curr)))))
  ;; emacs window title

        '("" invocation-name
          (:eval (when persp-mode (format "[%s]" (persp-name (persp-curr))))))))
